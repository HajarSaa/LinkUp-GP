// Tooltip rules

1- <Tooltip id={id} place={place} className={`custom-tooltip`} />
===> id (لازم العنصرين يكونوا عندهم نفس  الاسم ومثلا هنمشي بالشكل)
===> place (top , bottom , right , left)
===> className (ثابت)

2- <span className={`${styles.info_icon} ${className}`}
        data-tooltip-id={id}
        data-tooltip-content={text}
    >

===> data-tooltip-id={id} (نفس الاسم اللي فوق)
===> data-tooltip-content={text} (النص اللي هيظهر)


3- Example

   <>
      <span
        className={`${styles.info_icon} ${className}`}
        data-tooltip-id={id}
        data-tooltip-content={text}
      >
        <FiHelpCircle />
      </span>
      <Tooltip id={id} place={place} className={`custom-tooltip`} />
    </>
